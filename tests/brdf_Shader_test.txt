### Color Definitions ###

color green 0.1 0.75 0.1
color white 0.85 0.85 0.85
color black 0 0 0
color gray 0.5 0.5 0.5
color light_color 25 25 25 

### Shaders ###
# Cornell box walls using BRDF diffuse
brdf_shader floor_shader white 0.6
brdf_shader ceiling_shader white 0.6
brdf_shader back_shader white 0.6
brdf_shader left_shader red 0.6
brdf_shader right_shader green 0.6

# Test spheres with different roughness

brdf_shader sphere_mid gray 0.5
brdf_shader sphere_high gray 1.0

# Emissive light and background
flat_shader light_shader light_color 1

sphere 0 0.75 -1.5 0.25 light_shader
